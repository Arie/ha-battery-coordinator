#include <tunables/global>

profile battery-coordinator flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  file,
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay v3
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,

  # Legacy S6 paths (v2 compat)
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,

  # Runtime
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Python runtime + app code
  /usr/lib/python3*/** r,
  /app/** r,

  # Add-on config
  /data/** r,

  # Outbound network (TCP for HTTP/HTTPS, UDP for DNS)
  network inet tcp,
  network inet6 tcp,
  network inet udp,
  network inet6 udp,
}
